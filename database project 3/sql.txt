answers:

1- Select Max(salary) AS max_salary,min(salary) As min_salary
 from employee;

2-SELECT e_lname, dept_name, dept_id
FROM employee, department
WHERE e.dept_id = d.dept_id AND dept_id = 555;

3-SELECT dept_id, AVG(salary) AS avg_salary
FROM employees
GROUP BY dept_id
HAVING COUNT(*) > 5;

4-select count(*)nbofprod,cust_id
  from customer
  groupby cust_id;

5-SELECT *
FROM customer
WHERE cust_id NOT IN (SELECT e_id FROM employee);

6-SELECT dept_name, dept_id
FROM department
WHERE dept_id NOT IN (SELECT dept_id FROM employee);

7-SELECT e1.lname AS employee_name
FROM employees e1
WHERE (e1.job_title, e1.salary) IN 
      (SELECT e2.job_title, e2.salary 
       FROM employees e2 
       WHERE e2.employee_id = e1.manager_id);
8-SELECT *
FROM employee e
WHERE e_id NOT IN (SELECT DISTINCT e_id FROM works_on)
AND salary = (SELECT MIN(salary) FROM employee );

9-SELECT e_id, e_fname, dept_id

SELECT cust_id, cust_name
